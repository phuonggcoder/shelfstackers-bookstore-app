# Thông tin keystore release cho app Android

# Lệnh tạo keystore:
keytool -genkey -v -keystore release.keystore -alias mykeyalias -keyalg RSA -keysize 2048 -validity 10000

# Password (storePassword, keyPassword):
vuong123

# Key alias:
mykeyalias

# Thông tin certificate:
CN=vuong, OU=idk, O=petshop, L=hcm, ST=tan binh, C=84

# Đường dẫn file keystore:
android/app/release.keystore

# Cách lấy lại SHA1:
keytool -list -v -keystore android/app/release.keystore -alias mykeyalias -storepass vuong123 -keypass vuong123
